# Icon

由svg生成的图标，可由任意svg文件生成并自由添加到Icon的Library中。

# 基本用法

import {Icon} from 'react-native-maui';

<Icon name={'arrow-left'} size={50} color={'f8e0b0'} />

# 属性

名称 描述 类型 默认值
name
Required
 在Icon的Library中声明过的key string -
size 图标大小 number 20
color 图标填充颜色 string black
